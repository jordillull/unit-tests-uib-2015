import unittest

class MyTest(unittest.TestCase):

    def testToUpper(self):
        self.assertEqual(toUpper('Hello world'), 'HELLO WORLD')
            

def toUpper(str):
    new_str = ""
    for chr in str:
        new_str.append(chr.upper())

    return str
            

if __name__ == '_main__':
    unittest.main()
